# Ejercicio: Crear un vector que haga la diferencia entre variables cualitativas y cuantitativas.
dcuanti<-function(bd){
  bd<-data.frame(bd)
  aux<-NULL
  for(i in 1:ncol(bd)){
    aux[i]<-is.numeric(bd[,i])
  }
  return(aux)
}
